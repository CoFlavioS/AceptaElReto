#include <stdio.h>
 
#define gc getchar_unlocked
 
int scan_fast(){
	int n=0;
	char ch=gc();
	while(ch < '0' && ch >'9'){
		ch=gc(); // ignoring chars that are not digit
	}
	while(ch>='0' && ch<='9'){
		n=n*10+ch-'0'; // for each digit add it to n in proper place
		ch=gc();
	}
	return n;
}
 
void orden(int izq, int der, int a[]){
    int i,j,tmp,x;
    i = izq;
    j = der;
    x = a[(izq + der)/2];
    do{
        while (a[i] < x) i++;
        while (x < a[j]) j--;
        if (i <= j){
            tmp = a[i];
            a[i] = a[j];
            a[j] = tmp;
            i++;
            j--;
        }
    }while (i <= j);
    if (izq < j)
        orden(izq, j, a);
    if (i < der)
        orden(i, der, a);
}
 
int esRepdig(int num){
    int itis = 1, i = 1, aux, fir;
 
    fir = num % 10;
    num /= 10;
    do{
        aux = num % 10;
        num /= 10;
        if(fir != aux) itis = 0;
        i++;
    } while (itis && num!=0);
 
    return itis;
}
 
void intToVector(int num, int aux[]) {
    int i=0;
 
    do{
        aux[3-i] = num % 10;
        num /= 10;
        i++;
    } while (num != 0 || i < 4);
}
 
int main()
{
    int iter, pru, i, num, inv, aux[4];
 
    pru = scan_fast();
    while(pru-- > 0){
        num = scan_fast();
        if(esRepdig(num)){
            iter = 8;
        }
        else{
            iter = 0;
            while(num != 6174){
                iter++;
                intToVector(num, aux);
                orden(0,3,aux);
                inv = 0;
                num = 0;
                for(i = 0; i < 4; i++){
                    num = num*10 + aux[i];
                    inv = inv*10 + aux[3-i];
                }
                num = inv - num;
            }
        }
        printf("%d\n", iter);
    }
 
    return 0;
}
